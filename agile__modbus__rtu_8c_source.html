97 static uint16_t agile_modbus_rtu_crc16(uint8_t *buffer, uint16_t buffer_length)
98 {
99  uint8_t crc_hi = 0xFF; /* high CRC byte initialized */
100  uint8_t crc_lo = 0xFF; /* low CRC byte initialized */
101  unsigned int i; /* will index into CRC lookup */
102 
103  /* pass through message buffer */
104  while (buffer_length--) {
105  i = crc_hi ^ *buffer++; /* calculate the CRC */
106  crc_hi = crc_lo ^ _table_crc_hi[i];
107  crc_lo = _table_crc_lo[i];
108  }
109 
110  return (crc_hi << 8 | crc_lo);
111 }

199 static int agile_modbus_rtu_check_integrity(agile_modbus_t *ctx, uint8_t *msg, const int msg_length)
200 {
201  uint16_t crc_calculated;
202  uint16_t crc_received;
203 
204  crc_calculated = agile_modbus_rtu_crc16(msg, msg_length - 2);
205  crc_received = (msg[msg_length - 2] << 8) | msg[msg_length - 1];
206 
207  /* Check CRC of msg */
208  if (crc_calculated == crc_received)
209  return msg_length;
210 
211  return -1;
212 }
